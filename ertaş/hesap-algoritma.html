1. ARAÇ MALİYET ALGORİTMASI
// STOK sayfasından çıkarılan formüller
MASRAF = sigorta + noter + komisyon + yol_masrafi + kaporta + gocuk + muayene + diger
TOPLAM_MALİYET = alis_fiyati + MASRAF

// Örnekler (Excel'den):
MİTSUBİSHİ 34YL4249 → 583.000 + 13.330 = 596.330 ₺
HILUX 01FDL78 → 325.000 + 5.500 = 330.500 ₺
BMW 34PED073 → 4.130.000 + 0 = 4.130.000 ₺
2. NET KAR-ZARAR ALGORİTMASI
NET_KAR = satis_fiyati - TOPLAM_MALİYET

// Mart 2026 satılan araçlar örneği:
MEGANE 34BFJ584 → 1.120.000 - 977.800 = 142.200 ₺ KAR
AUDİA5 01BCR802 → 1.085.000 - 801.500 = 283.500 ₺ KAR
HILUX 09AGR986 → 1.560.000 - 1.476.690 = 83.310 ₺ KAR
MEGANE 34HAY131 → 1.007.000 - 1.007.000 = 0 ₺ (BAŞABAŞ)

// Mart 2026 toplam kar: 882.870 ₺
3. KİRALAMA GELİR ALGORİTMASI
TOPLAM_KİRA_GELİRİ = aylik_kira × kira_suresi_ay
KIRA_MASRAFI = sigorta + kasko + noter + takip_cihazi + kredi_masrafi
NET_KİRA_GELİRİ = TOPLAM_KİRA_GELİRİ - KIRA_MASRAFI
BİTİŞ_TARİHİ = baslangic + kira_suresi_ay (ay bazında)

YILLIK_GETİRİ_ORANI = (NET_KİRA_GELİRİ / 12 × 12) / araç_maliyet × 100

// Excel'den: TOPLAM GELEN = 601.000 ₺ (Mart 2026)
// GENEL TOPLAM MALİYET: 22.852.229 ₺ filo değeri
// TOPLAM KİRA MASRAF: 1.410.195 ₺ (sigorta+kasko+noter+takip+kredi)
4. ALACAK TAKİP ALGORİTMASI
KALAN_ALACAK = toplam_alacak - tahsil_edilen
TAHSILAT_ORANI = tahsil_edilen / toplam_alacak × 100

// Alacak türleri:
AÇIK_ALACAK → kişiden direkt borç (Görkem Çelik 165.000×4 senet)
ALINAN_ÇEK → vadeli çek (Soner Uçan: 135.000+155.000=290.000)
SENET → noter onaylı (Tolgahan Avcı 270.000, kalan 200.000)

// Excel toplam alacak: 6.862.347 ₺ (açık) + 290.000 ₺ (çek)
5. ORTAK CARİ HESAP ALGORİTMASI
ORTAK_BAKİYE = SUM(alacaklar) - SUM(borclar)
Bakiye > 0 → Şirketten alacaklı (A)
Bakiye < 0 → Şirkete borçlu (B)

// Excel'den Mart 2026 son bakiyeler:
Tamer YALINIZ → 1.191.000 - 1.553.790 = -362.790 (B)
Feramiz YALINIZ → 735.000 - 35.000 = +700.000 (A)
İlker YALINIZ → 1.380.000 - 645.160 = +734.840 (A)
Mehmet BAYDUMAN → 66.000 - 6.754 = +59.246 (A)
6. BİLANÇO ALGORİTMASI
DÖNEN_VARLIKLAR = stok + kasa + alinan_cekler + alacak_senetleri + acik_alacaklar - borclar

ORTAK_SERMAYE = özkaynak + (toplam_kar × pay_orani)

KAR_ORANI = (DÖNEN_VARLIKLAR - BILANÇO_TOPLAMI) / BILANÇO_TOPLAMI

// Excel NİSAN 2026:
Stok: 34.965.231 + Kasa: 1.517.434 + Çek: 125.000 + Senet: 782.500 = 37.390.165
Bilanço Toplamı (ortaklar): 40.298.068
Çıkan Fazlalık (kar dahil): 1.564.124
KAR_ORANI = 1.564.124 / 40.298.068 = %3.88
